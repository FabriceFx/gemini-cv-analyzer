### **1️⃣ Performances & Optimisations (Priorité ⭐⭐⭐⭐⭐)**
#### **Problèmes identifiés**
- **`fetchAll` dans `analyzeDocumentsBatch`** : Les appels parallèles à `UrlFetchApp.fetchAll` peuvent échouer silencieusement si un document est trop volumineux ou corrompu.
- **Taille des fichiers** : La limite de 20MB est vérifiée, mais pas la taille **totale du batch** (Gemini a une limite de tokens par requête).
- **Context Caching** : Le cache est créé pour tous les lots ≥3, mais la condition `fullText.length < 130000` est arbitraire (32k tokens ≈ 130k caractères, mais dépend du modèle).
- **`Utilities.sleep` bloquant** : Utilisé dans les boucles, ce qui ralentit l'exécution.

#### **Solutions proposées**
```javascript
// Dans Constants.gs
const MAX_BATCH_TOKENS = 150000; // Limite estimée pour gemini-3.5-flash
const MAX_TOTAL_TOKENS_PER_REQUEST = 200000; // À ajuster selon le modèle

// Dans GeminiClient.gs
function _prepareDocumentEntry(file) {
  const fileSizeBytes = file.getSize();
  if (fileSizeBytes > MAX_FILE_SIZE) {
    throw new Error(`Fichier trop volumineux (${Math.round(fileSizeBytes / 1024 / 1024)} MB). Max: 20 MB.`);
  }
  // Ajouter une estimation de tokens (1 token ≈ 4 caractères pour du texte)
  const estimatedTokens = fileSizeBytes / 4;
  if (estimatedTokens > MAX_TOTAL_TOKENS_PER_REQUEST) {
    throw new Error(`Fichier trop grand en tokens (${Math.round(estimatedTokens)}). Max: ${MAX_TOTAL_TOKENS_PER_REQUEST}.`);
  }
  // ...
}

// Dans analyzeDocumentsBatch (GeminiClient.gs)
function analyzeDocumentsBatch(files, apiKey, model, jobDescription, criteria, systemPrompt, cacheName) {
  const url = `https://generativelanguage.googleapis.com/v1beta/models/${model}:generateContent?key=${apiKey}`;
  const entries = [];
  const prepErrors = [];

  // Calculer la taille totale estimée du batch
  let totalEstimatedTokens = 0;
  for (const file of files) {
    try {
      const entry = _prepareDocumentEntry(file);
      totalEstimatedTokens += entry.base64Data.length / 4; // Estimation grossière
      entries.push(entry);

      prepErrors.push({ file, error: e.message });

  const subBatches = [];
  for (let i = 0; i < entries.length; i += SUB_BATCH_SIZE) {
    subBatches.push(entries.slice(i, i + SUB_BATCH_SIZE));

      method: "post",

      payload: JSON.stringify(_buildDocumentPayload(entry, jobDescription, criteria, systemPrompt, cacheName)),
      muteHttpExceptions: true
    }));

    const responses = requests.length ? UrlFetchApp.fetchAll(requests) : [];

  }
  return results;
```

### **2️⃣ Gestion des Erreurs & Robustesse (Priorité ⭐⭐⭐⭐⭐)**
#### **Problèmes identifiés**
- **`parseJsonSafely`** : Ne gère pas les cas où l'API retourne du texte non-JSON (ex: erreurs 429 en texte brut).
- **`fetchJobDescription`** : Les heuristiques pour détecter les pages SPA sont basiques (mots-clés en français seulement).
- **Pas de retry pour les erreurs 500/503** dans `callGeminiAPI` (seulement pour 429).
- **Pas de validation des emails** dans `draftEmailsForCandidates` (ex: `test@test` est accepté).

#### **Solutions proposées**
```javascript
// Dans Utils.gs
function parseJsonSafely(text) {
  if (!text || typeof text !== 'string') {
    throw new Error("Réponse vide ou invalide de l'API.");
  }

  let cleaned = text.trim();

  // Supprimer les balises Markdown
  cleaned = cleaned.replace(/^```(?:json)?\s*/i, "").replace(/```$/i, "").trim();

  // Extraire le premier objet JSON valide
  const jsonMatch = cleaned.match(/\{[^}]*(?:\{[^}]*\}[^}]*)*\}/);
  if (jsonMatch) {
    try {
      return JSON.parse(jsonMatch[0]);
    } catch (e) {
      Logger.log(`Échec du parsing JSON (match): ${e.message}`);
    }
  }

  // Essayer de parser directement
  try {
    return JSON.parse(cleaned);
  } catch (e) {
    Logger.log(`Échec du parsing JSON (direct): ${e.message}`);
    throw new Error(`Impossible de décoder la réponse IA. Texte reçu: "${cleaned.substring(0, 200)}..."`);
  }
}

// Dans callGeminiAPI (GeminiClient.gs)
function callGeminiAPI(model, payload, apiKey) {
  const url = `https://generativelanguage.googleapis.com/v1beta/models/${model}:generateContent?key=${apiKey}`;
  const options = {
    method: "post",
    contentType: "application/json",
    payload: JSON.stringify(payload),
    muteHttpExceptions: true
  };

  const maxRetries = 5; // Augmenté de 3 à 5
  let delay = 2500;
  let lastError = null;

  for (let attempt = 0; attempt < maxRetries; attempt++) {
    const response = UrlFetchApp.fetch(url, options);
    const code = response.getResponseCode();
    const text = response.getContentText();

    if (code === 200) {
      return text;
    }

    // Gérer les erreurs 500/503 comme les 429
    if (code === 429 || code === 500 || code === 503) {
      Logger.log(`API Gemini (${code}) - Tentative ${attempt + 1}/${maxRetries}. Attente: ${delay}ms`);
      Utilities.sleep(delay);
      delay *= 2;
      lastError = `Erreur ${code} après ${maxRetries} tentatives.`;
      continue;
    }

    // Extraire le message d'erreur
    let errorMsg = `Erreur HTTP ${code}`;
    try {
      const errJson = JSON.parse(text);
      if (errJson?.error?.message) {
        errorMsg = errJson.error.message;
      }
    } catch (e) {
      Logger.log(`Impossible de parser l'erreur: ${text.substring(0, 200)}`);
    }

    // Ne pas retry pour les erreurs 400/401/403 (problème client)
    if (code === 400 || code === 401 || code === 403) {
      throw new Error(errorMsg);
    }

    lastError = errorMsg;
  }

  throw new Error(lastError || "Échec après plusieurs tentatives.");
}

// Dans EmailService.gs
function isValidEmail(email) {
  return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
}

// Dans draftEmailsForCandidates
if (!isValidEmail(email)) {
  Logger.log(`Email invalide pour ${candidateName}: ${email}`);
  continue;
}
```

```javascript
const SUPPORTED_MIME_TYPES = [

// Dans Main.gs (ajouter une fonction pour analyser les sous-dossiers)
function getAllFilesInFolder(folder, supportedMimes) {
  const files = [];
  const subfolders = [];

  // Lister les fichiers du dossier courant
  const folderFiles = folder.getFiles();
  while (folderFiles.hasNext()) {
    const file = folderFiles.next();
    if (supportedMimes.includes(file.getMimeType())) {
      files.push(file);
    }
  }

  // Lister les sous-dossiers
  const folderIterator = folder.getFolders();
  while (folderIterator.hasNext()) {
    subfolders.push(folderIterator.next());
  }

  // Récursivité pour les sous-dossiers
  for (const subfolder of subfolders) {
    files.push(...getAllFilesInFolder(subfolder, supportedMimes));
  }

  return files;
}
```

```javascript
 * @param {string} systemPrompt
 * @param {string|null} cacheName
 * @returns {Array<{file: GoogleAppsScript.Drive.File, analysis: AnalysisResult, error?: string}>}
 */
function analyzeDocumentsBatch(files, apiKey, model, jobDescription, criteria, systemPrompt, cacheName) {
  const url = `https://generativelanguage.googleapis.com/v1beta/models/${model}:generateContent?key=${apiKey}`;
  const entries = [];
  const prepErrors = [];

  // Préparer les entrées et estimer la taille totale
  let totalEstimatedTokens = 0;
  for (const file of files) {
    try {
      const entry = _prepareDocumentEntry(file);
      totalEstimatedTokens += Math.ceil(entry.base64Data.length / 4); // 1 token ≈ 4 caractères
      entries.push(entry);
    } catch (e) {
      prepErrors.push({ file, error: e.message });
    }
  }

  if (entries.length === 0) {
    return prepErrors;
  }

  // Calculer la taille maximale par sous-batch
  const avgTokensPerFile = totalEstimatedTokens / entries.length;
  const maxFilesPerSubBatch = Math.max(
    1,
    Math.floor(MAX_BATCH_TOKENS / avgTokensPerFile)
  );

  // Diviser en sous-batches
  const subBatches = [];
  for (let i = 0; i < entries.length; i += maxFilesPerSubBatch) {
    subBatches.push(entries.slice(i, i + maxFilesPerSubBatch));
  }

  // Traiter chaque sous-batch
  const results = [...prepErrors];
  for (const subBatch of subBatches) {
    const requests = subBatch.map(entry => ({
      url,
      method: "post",
      contentType: "application/json",
      payload: JSON.stringify(_buildDocumentPayload(entry, jobDescription, criteria, systemPrompt, cacheName)),
      muteHttpExceptions: true
    }));

    const responses = UrlFetchApp.fetchAll(requests);
    for (let i = 0; i < responses.length; i++) {
      const response = responses[i];
      const entry = subBatch[i];
      const code = response.getResponseCode();

      if (code === 200) {
        try {
          results.push({
            file: entry.file,
            analysis: parseJsonSafely(_extractGeminiText(response.getContentText()))
          });
        } catch (e) {
          results.push({ file: entry.file, error: `Réponse illisible : ${e.message}` });
        }
      } else if (code === 429) {
        // Retry en solo
        try {
          const retryText = callGeminiAPI(
            model,
            _buildDocumentPayload(entry, jobDescription, criteria, systemPrompt, cacheName),
            apiKey
          );
          results.push({
            file: entry.file,
            analysis: parseJsonSafely(_extractGeminiText(retryText))
          });
        } catch (retryErr) {
          results.push({ file: entry.file, error: retryErr.message });
        }
      } else {
        let errorMsg = `Erreur HTTP ${code}`;
        try {
          const errJson = JSON.parse(response.getContentText());
          if (errJson?.error?.message) errorMsg = errJson.error.message;
        } catch (e) {}
        results.push({ file: entry.file, error: errorMsg });
      }
    }
    // Pause entre les sous-batches pour éviter le rate limiting
    Utilities.sleep(1000);
  }

  return results;
}
```
